# Aspect Ratio

Displays content within a desired ratio.

## Example

```tsx
import { AspectRatio } from '#/components/base-ui'

// 16:9 ratio
<AspectRatio ratio={16 / 9}>
  <img
    src="your-image.jpg"
    alt="Image description"
    className="object-cover w-full h-full"
  />
</AspectRatio>

// Square ratio (1:1)
<AspectRatio ratio={1}>
  <img
    src="your-image.jpg"
    alt="Image description"
    className="object-cover w-full h-full"
  />
</AspectRatio>

// Portrait ratio (3:4)
<AspectRatio ratio={3/4}>
  <img
    src="your-image.jpg"
    alt="Image description"
    className="object-cover w-full h-full"
  />
</AspectRatio>
```
